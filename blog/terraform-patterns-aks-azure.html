          - task: TerraformCLI@0
            displayName: "Terraform Plan"
            inputs:

              environmentServiceName: $(AZURE_SUBSCRIPTION)

              vars: |
                environment=$(ENVIRONMENT)
          - publish: $(System.DefaultWorkingDirectory)/infra/env/$(ENVIRONMENT)/tfplan
